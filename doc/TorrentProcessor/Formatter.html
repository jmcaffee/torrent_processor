def Formatter.pQueryResults(results)

  case @omode
    when :raw
      results.each do |r|
        p r
      end
      #puts results
      
    when :pretty
      results.each do |i|
        if( i.kind_of?(Array) )
          puts i.join( " | " )
        else
          puts i
        end
      end
  end
  
end
